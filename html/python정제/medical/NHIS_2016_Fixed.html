def ill(x):
    str = x[:3]
    de = re.findall(r'([A-Z])([_]?)(\d{2})?', str)
    for n, d, t in de:
        if t == '':
            t = 0
        if n == 'A' or n == 'B':
            return 'AB'
        elif n == 'C' or (n == 'D' and int(t) <= 48):
            return 'CD48'
        elif n == 'D':
            return 'D50'
        elif n == 'H':
            if int(t) < 60:
                return 'H59'
            elif int(t) >= 60:
                return 'H60'
        elif n == ('S' or 'T'):
            return 'ST'
        elif n == ('V' or 'Y'):
            return 'VY'
        elif n == (n.startswith('F')):
            return 'F'
        else:
            return n
        pass
    pass
